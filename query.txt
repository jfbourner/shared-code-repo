index="ent_payments_npa_fps"
    kafka_topic IN ("a","b","c")
    eventSource!="payment-saas-gateway-fps-outbound-sip"
| rename "record.key" AS record_key
| eval mac_version='data.metadata.macGeneratedKeyVersion'
| eval _time_main=_time
| fields record_key, mac_version, _time_main

| append
    [search index=lbg_payments_fpg_ocp_logs
        kubernetes.container_name="fpg-lineinterface-x"
        kubernetes.pod_name=*
    | rename "record.key" AS record_key
    | rename "kubernetes.pod_name" AS pod_name
    | fields record_key, pod_name]

| stats
    max(_time_main) AS _time
    max(mac_version) AS mac_version
    max(pod_name) AS pod_name
    by record_key

| sort _time
| streamstats window=1 current=false last(mac_version) AS prev_version by pod_name
| eval version_changed = case(
    isnull(prev_version),         "first seen",
    mac_version != prev_version,  "CHANGED",
    true(),                       "unchanged"
  )
| where version_changed="CHANGED" OR version_changed="first seen"
| table _time, pod_name, prev_version, mac_version, version_changed
